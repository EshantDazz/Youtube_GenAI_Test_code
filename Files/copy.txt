from flask import Flask, render_template, url_for, request
import stripe
from config import pk,sk,ppk

app = Flask(__name__)

# Replace with your own Stripe API keys
stripe.api_key = pk
stripe_public_key = sk

@app.route('/')
def index():
    return render_template('index.html', stripe_public_key=stripe_public_key)

@app.route('/payment', methods=['POST'])
def payment():
    # Retrieve the payment token and other details from the form
    token = request.form['stripeToken']
    amount = 1000  # Amount in cents (10 USD)

    try:
        # Create a charge using the Stripe API
        charge = stripe.Charge.create(
            amount=amount,
            currency='usd',
            description='Payment',
            source=token
        )
        return render_template('thanks.html', amount=amount/100)  # Display a thank you page
    except stripe.error.CardError as e:
        # Display an error message if the payment is declined
        return render_template('error.html', error_message=e.user_message)

if __name__ == '__main__':
    app.run(debug=True)
